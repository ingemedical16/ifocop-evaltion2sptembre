/**
 * une function qui transforme les argument a une objet dont les key sont
 * l'index de l'argument
 * @param  {...any} args 
 * @returns {Object}
 */
const returnAnObject = (...args) => {
  let response = {};
  if (args.length) {
    let index = 0;
    args.forEach(arg => {
      response[index] = arg;
      index++;
    })
  } else {
    response = 'No argument was given to the function.';
  };
  return response;
};
